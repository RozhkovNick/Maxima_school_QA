import java.util.Scanner;
public class Lain {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Введите три стороны треугольника: ");
        int one = sc.nextInt();
        int two = sc.nextInt();
        int three = sc.nextInt();
        if (one == two &&  two == three){System.out.println ("Треугольник равносторонний");}
        else if (one == two || two == three || one == three )  {System.out.println ("Треугольник равнобедренный");}
        if (one != two &&  one != three &&  two != three) {System.out.println ("Треугольник разносторонний");}
    }
}
