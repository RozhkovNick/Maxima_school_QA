        int two = sc.nextInt();
        int three = sc.nextInt();
        int four = sc.nextInt();
        int x = 0;
        int y = 0;
        if(one > 0) x = x + 1;
        else y = y + 1;
        if(two > 0) x = x + 1;
        else y = y + 1;
        if(three > 0) x = x + 1;

}
